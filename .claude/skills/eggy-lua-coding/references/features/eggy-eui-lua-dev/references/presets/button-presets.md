## 使用示例

```lua
local tf = math.tofixed
local UINodes = require("Data.UINodes")
local canvas = UINodes["画布0"]

-- 创建确认按钮
local btnConfirm = GameAPI.create_eui_button_at_position(
    11001,  -- 金色按钮样式0
    canvas,
    tf(960), tf(400),
    tf(200), tf(80),
    "btn_confirm"
)

-- 创建取消按钮
local btnCancel = GameAPI.create_eui_button_at_position(
    11002,  -- 蓝色按钮样式0
    canvas,
    tf(960), tf(300),
    tf(200), tf(80),
    "btn_cancel"
)

-- 创建关闭按钮
local btnClose = GameAPI.create_eui_button_at_position(
    10005,  -- 关闭
    canvas,
    tf(1200), tf(700),
    tf(80), tf(65),
    "btn_close"
)

-- 注册点击事件
local TOUCH_CLICK = 1
LuaAPI.global_register_trigger_event(
    {EVENT.EUI_NODE_TOUCH_EVENT, btnConfirm, TOUCH_CLICK},
    function()
        print("确认按钮被点击")
    end
)
```
